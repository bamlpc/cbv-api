Queries

find_all_cbv

Response

Returns [StoreCBV]

Example

Query
query find_all_cbv {
  find_all_cbv {
    _id
    cbv {
      title
      short_description
      cbv_id
      blockchain
      version_affected
      component
      severity
      score
      vulnerability_type
      details
      recommendation
      references
      labels
      tests
      aditional_comments
      credits
      created_at
      updated_at
    }
    timestamp
  }
}
Response
{
  "data": {
    "find_all_cbv": [
      {
        "_id": "abc123",
        "cbv": CBV,
        "timestamp": 123.45
      }
    ]
  }
}
